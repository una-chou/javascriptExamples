        function sortArr ( arr ) {
            for(var i = 0; i < arr.length - 1; i++){
                var flag = true;
                for(var j = 0; j < arr.length - 1 - i; j++){
                    var temp;
                    if(arr[j] > arr[j + 1]){
                        temp = arr[j + 1];
                        arr[j + 1] = arr[j];
                        arr[j] = temp;
                        flag = false;
                    }
                }
                if(flag) break;
            }
            return arr;
        }
